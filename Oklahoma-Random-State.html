Weekly and Monthly Cases and Deaths

df$Month <- as.Date(cut(df$date,
  breaks = "month"))
df$Week <- as.Date(cut(df$date,
  breaks = "week",
  start.on.monday = FALSE)) # changes weekly break point to Sunday
weekly_cases <- aggregate(DailyCases~Week,data=df,FUN=sum)
weekly_deaths <- aggregate(DailyDeaths~Week,data=df,FUN=sum)
plot(weekly_cases$Week,weekly_cases$DailyCases,type="h",main="Weekly Cases")
grid(col="black")

plot(weekly_deaths$Week,weekly_deaths$DailyDeaths,type="h",main="Weekly Deaths")
grid(col="black")

monthly_cases <- aggregate(DailyCases~Month,data=df,FUN=sum)
monthly_deaths <- aggregate(DailyDeaths~Month,data=df,FUN=sum)
plot(monthly_cases$Month,monthly_cases$DailyCases,type="h",main="Monthly Cases")
grid(col="black")

plot(monthly_deaths$Month,monthly_deaths$DailyDeaths,type="h",main="Monthly Deaths")
grid(col="black")

barplot(DailyCases ~Week,data=weekly_cases,main="Weekly COVOD-19 Cases",xlab="Weekly Date",ylab="Cases")
grid(col="black")

barplot(DailyDeaths ~Week,data=weekly_deaths,main="Weekly COVID-19 Deaths",
        xlab="Weekly Date",ylab="Deaths")
grid(col="black")
