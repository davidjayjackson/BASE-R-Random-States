Total Cases, Total Deaths (w/ 14 Day Moving Avearge) by Date

RI$MAC <- movingAverage(RI$new_cases,14)
RI$MAD <- movingAverage(RI$new_deaths,14)
plot(RI$date,RI$new_cases,type="l",main="Total Cases by Date",col="black")
lines(RI$date,RI$MAC,col="red")
grid(col="black")

plot(RI$date,RI$new_deaths,type="l",main="Total Deaths by Date")
lines(RI$date,RI$MAD,col="red")
grid(col="black")

Weekly and MOntly New Cases and New Deaths.

RI$Monthly <- as.Date(cut(RI$date,
  breaks = "month"))
RI$Weekly <- as.Date(cut(RI$date,
  breaks = "week",
  start.on.monday = FALSE)) # changes weekly break point to Sunday
weekly_cases <- aggregate(new_cases~Weekly,data=RI,FUN=sum)
weekly_deaths <- aggregate(new_deaths~Weekly,data=RI,FUN=sum)
weekly_cases$RCases <- weekly_deaths$new_deaths / weekly_cases$new_cases *100
barplot(new_cases ~Weekly,data=weekly_cases,main="Weekly COVOD-19 Cases",xlab="Weekly Date",ylab="Cases")
grid(col="black")

barplot(new_deaths ~Weekly,data=weekly_deaths,main="Weekly COVID-19 Deaths",
        xlab="Weekly Date",ylab="Deaths")
grid(col="black")

barplot(RCases ~Weekly,data=weekly_cases,main="Weekly COVID-19 Deaths",
        xlab="Weekly Date",ylab="Deaths")
grid(col="black")
